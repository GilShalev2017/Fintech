import express from 'express';
import { v4 as uuidv4 } from 'uuid';

const app = express();
app.use(express.json());

// ─────────────────────────────────────────────────────────────
// In-memory stores (replace with Redis + PostgreSQL in production)
// ─────────────────────────────────────────────────────────────
const accounts = new Map([
  ['checking-123', 1000],   // initial balance
  ['savings-456',   500],
]);

const idempotencyStore = new Map(); // key → { status, result, timestamp }

// For demo: simulate network delay
const simulateDelay = (ms = 300) => new Promise(r => setTimeout(r, ms));

// ─────────────────────────────────────────────────────────────
// POST /transactions  ←  THE MONEY ENDPOINT
// ─────────────────────────────────────────────────────────────
app.post('/transactions', async (req, res) => {
  const idempotencyKey = req.headers['idempotency-key'];

  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotency-key header is required' });
  }

  console.log(`[REQUEST] Idempotency-Key: ${idempotencyKey}`);

  // 1. Check if we already processed this request
  const existing = idempotencyStore.get(idempotencyKey);
  if (existing) {
    console.log(`[IDEMPOTENT] Duplicate detected → returning cached result`);
    return res.status(existing.result.statusCode).json(existing.result.body);
  }

  const { fromAccount, toAccount, amount, description } = req.body;

  // Basic validation
  if (!fromAccount || !toAccount || !amount || amount <= 0) {
    return res.status(400).json({ error: 'Invalid transaction data' });
  }

  // 2. Start "atomic" operation (in real app → DB transaction)
  try {
    await simulateDelay(); // simulate DB/network

    const fromBalance = accounts.get(fromAccount) ?? 0;
    if (fromBalance < amount) {
      throw new Error('Insufficient funds');
    }

    // === CRITICAL: Do the money movement ===
    accounts.set(fromAccount, fromBalance - amount);
    accounts.set(toAccount, (accounts.get(toAccount) ?? 0) + amount);

    const transactionId = uuidv4();

    const successResponse = {
      statusCode: 201,
      body: {
        success: true,
        transactionId,
        fromAccount,
        toAccount,
        amount,
        description: description || 'Transfer',
        newFromBalance: accounts.get(fromAccount),
        newToBalance: accounts.get(toAccount),
        message: 'Transaction completed successfully'
      }
    };

    // 3. Record idempotency (only after successful processing)
    idempotencyStore.set(idempotencyKey, {
      status: 'completed',
      result: successResponse,
      timestamp: Date.now()
    });

    console.log(`[SUCCESS] Transaction ${transactionId} processed`);
    res.status(201).json(successResponse.body);

  } catch (error) {
    console.error('[ERROR]', error.message);

    // Still store failed attempts so retries get the same error
    idempotencyStore.set(idempotencyKey, {
      status: 'failed',
      result: {
        statusCode: 400,
        body: { success: false, error: error.message }
      },
      timestamp: Date.now()
    });

    res.status(400).json({ success: false, error: error.message });
  }
});

// Bonus: see current balances
app.get('/balances', (req, res) => {
  res.json(Object.fromEntries(accounts));
});

const PORT = 3000;
app.listen(PORT, () => {
  console.log(`\n🚀 Money API running on http://localhost:${PORT}`);
  console.log('   Try the same idempotency-key twice → should be safe!\n');
});

✅ Full Production-Style Solution
Here’s exactly what top fintech/Backend interviews expect for the “Money / transactions” question.

First – Your Two Questions Answered Clearly

1. What if the idempotency key is removed from Redis (TTL expiration)?
Redis is only a fast cache (TTL = 24h or 7 days).
The source of truth is the PostgreSQL table.

When a request arrives:

  1. Check Redis → hit = return instantly (fast path)
  2. Redis miss → query PostgreSQL for the idempotency_key
  3. If found in DB → return the saved result (exactly the same as before)
  4. If not in DB → process the transaction atomically and save to both DB + Redis.

→ Even if the Redis key was deleted 2 years ago, the server will still never duplicate the money movement.

2. Why does the client have to carry responsibility?
The server is 100% idempotent and safe by design.
The client’s only job is:

  • Generate one unique idempotency-key per logical transaction (usually a UUID created on the client).
  • On network timeout / 5xx / connection error → retry with the exact same key.
  • On business error (insufficient funds, etc.) → do not retry with same key (treat as final).

The server does not trust the client. It just guarantees:
“Same key = same result, no matter how many times you call it.”

This is exactly how Stripe, PayPal, Shopify, etc. work.

Full Server Code (server.js)

import express from 'express';
import { createClient } from 'redis';
import pkg from 'pg';
const { Client } = pkg;
import { v4 as uuidv4 } from 'uuid';

const app = express();
app.use(express.json());

const redis = createClient({ url: 'redis://localhost:6379' });
await redis.connect();

const db = new Client({
  host: 'localhost',
  port: 5432,
  user: 'postgres',
  password: 'secret',
  database: 'money'
});
await db.connect();

// TTL = 24 hours
const REDIS_TTL = 24 * 60 * 60;

app.post('/transactions', async (req, res) => {
  const idempotencyKey = req.headers['idempotency-key'];
  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotency-key header is required' });
  }

  console.log(`[REQUEST] Idempotency-Key: ${idempotencyKey}`);

  // 1. Fast path: Redis
  let cached = await redis.get(`idemp:${idempotencyKey}`);
  if (cached) {
    const data = JSON.parse(cached);
    console.log('[REDIS HIT] Returning cached result');
    return res.status(data.statusCode).json(data.body);
  }

  // 2. Check permanent record in PostgreSQL
  const { rows } = await db.query(
    'SELECT response_json FROM transactions WHERE idempotency_key = $1',
    [idempotencyKey]
  );

  if (rows.length > 0) {
    const data = rows[0].response_json;
    // Re-cache in Redis
    await redis.set(`idemp:${idempotencyKey}`, JSON.stringify(data), { EX: REDIS_TTL });
    console.log('[DB HIT] Returning permanent result');
    return res.status(data.statusCode).json(data.body);
  }

  // 3. New request → process atomically
  const { fromAccount, toAccount, amount, description } = req.body;

  try {
    await db.query('BEGIN');

    // Check balance
    const { rows: balanceRows } = await db.query(
      'SELECT balance FROM accounts WHERE id = $1 FOR UPDATE',
      [fromAccount]
    );

    if (balanceRows.length === 0 || parseFloat(balanceRows[0].balance) < amount) {
      throw new Error('Insufficient funds');
    }

    // Move money
    await db.query(
      'UPDATE accounts SET balance = balance - $1 WHERE id = $2',
      [amount, fromAccount]
    );
    await db.query(
      'UPDATE accounts SET balance = balance + $1 WHERE id = $2',
      [amount, toAccount]
    );

    const transactionId = uuidv4();
    const successBody = {
      success: true,
      transactionId,
      fromAccount,
      toAccount,
      amount,
      description: description || 'Transfer',
      message: 'Transaction completed successfully'
    };

    // Save permanently with UNIQUE constraint
    await db.query(
      `INSERT INTO transactions 
       (idempotency_key, from_account, to_account, amount, description, status, response_json)
       VALUES ($1, $2, $3, $4, $5, 'completed', $6)`,
      [idempotencyKey, fromAccount, toAccount, amount, description, { ...successBody, statusCode: 201 }]
    );

    await db.query('COMMIT');

    // Cache in Redis
    const cachePayload = { statusCode: 201, body: successBody };
    await redis.set(`idemp:${idempotencyKey}`, JSON.stringify(cachePayload), { EX: REDIS_TTL });

    console.log(`[SUCCESS] Transaction ${transactionId} saved`);
    res.status(201).json(successBody);

  } catch (err) {
    await db.query('ROLLBACK');

    const errorBody = { success: false, error: err.message };
    const cachePayload = { statusCode: 400, body: errorBody };

    // Still store failed attempts (so future retries get same error)
    await redis.set(`idemp:${idempotencyKey}`, JSON.stringify(cachePayload), { EX: REDIS_TTL });

    await db.query(
      `INSERT INTO transactions 
       (idempotency_key, from_account, to_account, amount, description, status, response_json)
       VALUES ($1, $2, $3, $4, $5, 'failed', $6)
       ON CONFLICT (idempotency_key) DO NOTHING`,
      [idempotencyKey, fromAccount || '', toAccount || '', amount || 0, description || '', { ...errorBody, statusCode: 400 }]
    );

    res.status(400).json(errorBody);
  }
});

// Bonus endpoint
app.get('/balances', async (req, res) => {
  const { rows } = await db.query('SELECT * FROM accounts');
  res.json(rows);
});

const PORT = 3000;
app.listen(PORT, () => {
  console.log(`🚀 Production Money API running on http://localhost:${PORT}`);
  console.log('   Redis + PostgreSQL idempotency enabled\n');
});

2. Client-Side Retry Script (Safe Retries)

// retry-client.js
import fetch from 'node-fetch';
import { v4 as uuidv4 } from 'uuid';

const IDEMPOTENCY_KEY = uuidv4();   // ← Generate ONCE per logical transaction

async function sendTransaction(attempt = 1) {
  console.log(`Attempt ${attempt} | Key: ${IDEMPOTENCY_KEY}`);

  try {
    const res = await fetch('http://localhost:3000/transactions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'idempotency-key': IDEMPOTENCY_KEY
      },
      body: JSON.stringify({
        fromAccount: 'checking-123',
        toAccount: 'savings-456',
        amount: 100,
        description: 'Rent'
      })
    });

    const data = await res.json();

    if (res.status === 201) {
      console.log('✅ SUCCESS:', data);
      return data;
    }

    // Business error (e.g. insufficient funds) → do NOT retry
    console.log('❌ Business error:', data);
    return data;

  } catch (err) {
    // Network / timeout / 5xx → retry with SAME key
    if (attempt < 5) {
      console.log(`⚠️ Network error → retrying in ${attempt * 500}ms...`);
      await new Promise(r => setTimeout(r, attempt * 500));
      return sendTransaction(attempt + 1);
    }
    console.log('💥 Max retries reached');
    throw err;
  }
}
